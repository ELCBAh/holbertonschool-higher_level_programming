class MyList(list):
    """
    A class inheriting from list
    """
    def __init__(self):
        super().__init__()

    def print_sorted(self):
        """
        prints the list in asc order

		Example:
        >>> my_list = MyList()
        >>> my_list.append(5)
        >>> my_list.append(2)
        >>> my_list.append(8)
        >>> my_list.print_sorted()
        [2, 5, 8]
        """
        print(sorted(self))
